function EquipItem(itemOID: cardinal; newState: boolean = true): Boolean;
var item: TL2Item;
begin

  if(item.Equipped <> newState) then
  begin
    Engine.useItem(item);
Result := true;
  end else
  begin
  Result := false;
  end;
end;
